use Bio::EnsEMBL::Registry;Bio::EnsEMBL::Registry->load_registry_from_db(    -host => 'ensembldb.ensembl.org',    -user => 'anonymous',    -port => 3306);

use Bio::EnsEMBL::Compara::DBSQL::DBAdaptormy $host = 'ensembldb.ensembl.org';my $user = 'anonymous';my $dbname = 'ensembl_compara_39';my $comparadb= new Bio::EnsEMBL::Compara::DBSQL::DBAdaptor(    -host	=> $host,    -user	=> $user,    -dbname => $dbname);

# get the MemberAdaptormy $member_adaptor = Bio::EnsEMBL::Registry->get_adaptor(    'Multi','compara','Member');

#Variables
my $geneId = 'ENSG00000004059';
my $orthoGeneId = $geneId; //default is to keep ID if no ortho match
# fetch a Membermy $member = $member_adaptor->fetch_by_source_stable_id(    'ENSEMBLGENE', $geneId);

# then you get the homologies where the member is involvedmy $homology_adaptor = Bio::EnsEMBL::Registry->get_adaptor('Multi', 'compara', 'Homology');my $homologies = $homology_adaptor->fetch_all_by_Member_paired_species($member, 'Mus musculus');
# That will return a reference to an array with all homologies
# Then for each homology, you can get all the Members implicatedforeach my $homology (@{$homologies}) {
  if ($homology->subtype = 'URBH') {	my $pair = $homology->gene_list();
	foreach my $memberGene (@{$pair}) {
 	   if ($memberGene != $member) { // not self
		my $orthoGene = $memberGene->get_Gene();
		$orthoGeneId = $orthoGene->stable_id();
	    }
	}
  }
}

USAGE:
$orthoGeneId replaces $geneId in GPML
